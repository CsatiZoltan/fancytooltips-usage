      \label{alg:testAlg}
      \begin{algorithmic}[1]
         \State $N = 10$ \Comment{Initialise loop count}
         \State \texttt{fac} $\gets 1$ \Comment{Set starting value, i.e. $0!$}
         \For {$i$ from 1 to $N$}
            \State \texttt{fac} $\gets$ \texttt{fac}*$i$
         \EndFor
      \end{algorithmic}
   \end{algorithm}
